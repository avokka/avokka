Case class generator

With the help of magnolia, we can instantiate codecs for case classes as velocypack objects (VPackDecoder.gen), or arrays (VPackGeneric) :

import avokka.velocypack._
import scodec.bits._

case class Test(b: Boolean)
implicit val testEncoder: VPackEncoder[Test] = VPackEncoder.gen
// testEncoder: VPackEncoder[Test] = avokka.velocypack.VPackEncoder$$anon$2@109db9a
implicit val testDecoder: VPackDecoder[Test] = VPackDecoder.gen
// testDecoder: VPackDecoder[Test] = avokka.velocypack.VPackDecoder$$anon$1@5af026fe

Encoding :

val t = Test(true)
// t: Test = Test(true)

t.toVPack      
// res0: VPack = VObject(Map(b -> VTrue))      
t.toVPackBits
// res1: VPackResult[BitVector] = Right(BitVector(56 bits, 0x0b070141621a03))

Decoding :

hex"0b070141621903".bits.asVPack[Test]
// res2: VPackResult[scodec.DecodeResult[Test]] = Right(DecodeResult(Test(false),BitVector(empty)))
hex"0a".bits.asVPack[Test]                                                                                         
// res3: VPackResult[scodec.DecodeResult[Test]] = Left(avokka.velocypack.VPackError$ObjectFieldAbsent: object field absent b)

Using defaults :

case class TestTrue(b: Boolean = true)                

implicit val testTrueDecoder: VPackDecoder[TestTrue] = VPackDecoder.gen
// testTrueDecoder: VPackDecoder[TestTrue] = avokka.velocypack.VPackDecoder$$anon$1@76525cd5

hex"0b070141621903".bits.asVPack[TestTrue]            
// res4: VPackResult[scodec.DecodeResult[TestTrue]] = Right(DecodeResult(TestTrue(false),BitVector(empty)))            
hex"0a".bits.asVPack[TestTrue]                                                                                                         
// res5: VPackResult[scodec.DecodeResult[TestTrue]] = Right(DecodeResult(TestTrue(true),BitVector(empty)))
